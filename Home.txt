import java.text.NumberFormat;

public class Home extends Policy {
	
	private double contents, dwelling, liability;
	private int footage;
	
	public Home()
	{
		firstName = " ";
		lastName = " ";
		contents = 0;
		dwelling = 0;
		liability = 0;
		footage = 0;
		
	}
	
	public Home(String first, String last, double c, double d, int f, double l)
	{
		firstName = first;
		lastName = last;
		contents = c;
		dwelling = d;
		liability = l;
		footage = f;
		
	}
	
	//setters
	
	public void setContents(double co)
	{
			contents = co;
	}
	
	public void setDwelling(double dw)
	{
			dwelling = dw;
	}
	
	public void setLiability(double li)
	{
			liability = li;
	}
	
	public void setFootage(int fo)
	{
			footage = fo;
	}
	
	
	
	//getters
	
	public double getContents()
	{
			return contents;
	}
	
	public double getDwelling()
	{
			return dwelling;
	}
	
	public double getLiability()
	{
			return liability;
	}
	
	public int getFootage()
	{
			return footage;
	}
	
	//tostring
	
	public String toString()
	{
	String result;
	
	NumberFormat fmt1 = NumberFormat.getCurrencyInstance();
	
	result = "\n\tHome Policy" + "\n\t";
	result += "----------" + "\n\t";
	result += "Name: " + firstName + " " + lastName + "\n\t";
	result += "Footage: " + footage + "\n\t";
	result += "Dwelling: " + fmt1.format(dwelling) + "\n\t";
	result += "Liability: " + fmt1.format(liability) + "\n\t";
	result += "Contents: " + fmt1.format(contents) + "\n\t";
	result += "Commission: " + fmt1.format(commission) + "\n";
	
	return result;
	}
	
	
	//commission calculation
	public void computeCommission()
	{
		commission = liability * 0.30 + (dwelling + contents) * 0.20;
	}
	
}
