import java.text.NumberFormat;

public class Life extends Policy {
	
	private double term;
	private int age;
	
	
	public Life()
	{
		firstName = " ";
		lastName = " ";
		age = 0;
		term = 0;
		
	}
	
	public Life(String first, String last, int a, double t)
	{
		firstName = first;
		lastName = last;
		age = a;
		term = t;
		
	}
	
	//setters
	
	public void setAge(int ag)
	{
			age = ag;
	}
	
	public void setTerm(double te)
	{
			term = te;
	}
	
	
	
	//getters
	
	public int getAge()
	{
			return age;
	}
	
	public double getTerm()
	{
			return term;
	}
	
	//tostring
	
	public String toString()
	{
	String result;
	
	NumberFormat fmt1 = NumberFormat.getCurrencyInstance();
	
	result = "\n\tLife Policy" + "\n\t";
	result += "----------" + "\n\t";
	result += "Name: " + firstName + " " + lastName + "\n\t";
	result += "Age: " + age + "\n\t";
	result += "Term: " + fmt1.format(term) + "\n\t";
	result += "Commission: " + fmt1.format(commission) + "\n";
	
	return result;
	}
	
	
	//commission calculation
	public void computeCommission()
	{
		commission = term * 0.20;
	}
	
}
